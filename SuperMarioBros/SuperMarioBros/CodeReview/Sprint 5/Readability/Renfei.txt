                x = -marioWidth/2;

                b = marioHeight * 0.5f;
                c = 0;
            }
            else if (pX >= worldWidth * objectWidth - marioWidth/2)
            {
                x = -worldWidth * objectWidth + marioWidth / 2;
                y = marioHeight / 2 - 256;
                z = 0;

                b = marioHeight * 0.5f;

            //Matrix Camera = Matrix.CreateTranslation(x, y, z);
